* Handshake

#+BEGIN_SRC plantuml :file PicoTalk-handshake.svg
  @startuml
  concise "Status" as Status
  robust "Transmitter" as TX
  robust "Receiver" as RX

  Status is Idle
  TX is 0
  RX is 0

  @0

  @+1
  Status is "Failing handshake (no RX)"
  TX is 1

  @+2
  TX is 0

  @+4
  Status is Handshake
  TX is 1

  @+2
  TX is 0

  @+1
  RX is 1

  @+1
  RX is 0

  @+2
  Status is "Preamble (see below)"
  TX is 1

  @enduml
#+END_SRC
